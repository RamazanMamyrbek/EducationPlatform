CREATE TABLE USERS(    id int PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY ,
                       first_name varchar(30) NOT NULL ,    last_name varchar(30) NOT NULL ,
                       email varchar(30) NOT NULL ,    password varchar(150) NOT NULL ,
                       phone_number varchar(30) NOT NULL,    role varchar(30) not null default 'STUDENT'
);
drop table users;
CREATE TABLE COURSES(
                        id int PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY ,
                        name varchar(30) NOT NULL ,
                        price int,
                        link varchar(200)
);

CREATE TABLE users_courses(
                              user_id INT REFERENCES USERS(id),
                              course_id INT REFERENCES COURSES(id),
                              PRIMARY KEY (user_id, course_id)
);
